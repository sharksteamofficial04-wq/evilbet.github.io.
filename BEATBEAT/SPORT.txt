    <script>
        document.addEventListener('DOMContentLoaded', () => {
            // Инициализация купона
            initCoupon();
            updateBalance();
            highlightSelectedOdds();

            // Обработчик клика по коэффициентам
            document.querySelectorAll('.odds-item').forEach(item => {
                item.addEventListener('click', function(e) {
                    e.stopPropagation();
                    const event = this.getAttribute('data-event');
                    const bet = this.getAttribute('data-bet');
                    const odds = this.getAttribute('data-odds');
                    const matchId = this.getAttribute('data-match-id');
                    const added = addToCoupon(event, bet, odds, matchId);
                    if (added) {
                        this.classList.add('selected');
                        document.querySelector('.bet-coupon').classList.add('visible');
                        showNotification(`Добавлено в купон: ${event} - ${bet} (${odds})`);
                    } else {
                        removeFromCouponByEvent(event, bet, matchId);
                        this.classList.remove('selected');
                        if (!isCouponNotEmpty()) {
                            document.querySelector('.bet-coupon').classList.remove('visible');
                        }
                    }
                });
            });

            // Обработчик вкладок купона
            document.querySelectorAll('.coupon-tab').forEach(tab => {
                tab.addEventListener('click', () => {
                    document.querySelectorAll('.coupon-tab').forEach(t => t.classList.remove('active'));
                    tab.classList.add('active');
                    const tabId = tab.getAttribute('data-tab');
                    document.getElementById('couponItems').style.display = tabId === 'bets' ? 'block' : 'none';
                    document.getElementById('couponHistory').style.display = tabId === 'history' ? 'block' : 'none';
                    document.querySelector('.coupon-total').style.display = tabId === 'bets' ? 'block' : 'none';
                    document.querySelector('.coupon-header').style.display = tabId === 'bets' ? 'flex' : 'none';
                    if (tabId === 'history') {
                        updateCouponHistory();
                    }
                });
            });

            // Обработчик выбора вида спорта
            document.querySelectorAll('.sport-item').forEach(item => {
                item.addEventListener('click', function() {
                    document.querySelectorAll('.sport-item').forEach(i => i.classList.remove('active'));
                    this.classList.add('active');
                    const sportName = this.querySelector('.sport-name').textContent;
                    document.querySelector('.section-title').textContent = `${sportName} МАТЧИ`;
                });
            });

            // Обработчик навигации по датам
            document.querySelectorAll('.date-arrow').forEach(arrow => {
                arrow.addEventListener('click', function() {
                    const currentTitle = document.querySelector('.date-title').textContent;
                    console.log('Переключение даты:', currentTitle);
                });
            });

            // Обработчик уведомлений
            document.querySelector('.notifications-btn').addEventListener('click', () => {
                showNotification('Уведомления пока не реализованы');
                document.querySelector('.notification-count').textContent = '0';
            });
        });

        // Модалка профиля
        function openProfile() {
            document.getElementById('profileModal').classList.add('show');
            updateBetHistory();
        }

        function closeProfile() {
            document.getElementById('profileModal').classList.remove('show');
        }

        // Модалка истории ставок
        function openHistory() {
            document.getElementById('profileModal').classList.remove('show');
            document.getElementById('historyModal').classList.add('show');
            updateBetHistory();
        }

        function closeHistory() {
            document.getElementById('historyModal').classList.remove('show');
        }
    </script>
