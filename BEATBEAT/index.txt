    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const betCoupon = document.querySelector('.bet-coupon');
            const clearCouponBtn = document.querySelector('.clear-coupon');
            const userName = document.getElementById('userName');
            const connectTgBtn = document.getElementById('connectTgBtn');
            const notificationCount = document.querySelector('.notification-count');
            const notificationsBtn = document.querySelector('.notifications-btn');

            const userNameText = 'Гость';
            userName.textContent = userNameText;
            document.getElementById('profileName').textContent = userNameText;
            connectTgBtn.style.display = 'inline-block';

            document.querySelectorAll('.match-card').forEach(card => {
                card.addEventListener('click', (e) => {
                    if (e.target.closest('.odds-item')) {
                        return;
                    }
                    const matchId = card.getAttribute('data-match-id');
                    const matchTime = card.querySelector('.match-time').textContent;
                    const league = card.querySelector('.match-league').textContent;
                    const teamHome = card.querySelector('.team-home .team-name').textContent;
                    const teamAway = card.querySelector('.team-away .team-name').textContent;
                    localStorage.setItem('selectedMatch', JSON.stringify({
                        id: matchId,
                        time: matchTime,
                        league: league,
                        teamHome: teamHome,
                        teamAway: teamAway
                    }));
                    window.location.href = 'match-events.html';
                });
            });

            document.querySelectorAll('.odds-item').forEach(item => {
                item.addEventListener('click', function(e) {
                    e.stopPropagation();
                    const event = this.getAttribute('data-event');
                    const bet = this.getAttribute('data-bet');
                    const odds = this.getAttribute('data-odds');
                    const matchId = this.getAttribute('data-match-id');
                    const added = addToCoupon(event, bet, odds, matchId);
                    if (added) {
                        this.classList.add('selected');
                        betCoupon.classList.add('visible');
                        showNotification(`Добавлено в купон: ${event} - ${bet} (${odds})`);
                    } else {
                        removeFromCouponByEvent(event, bet, matchId);
                        this.classList.remove('selected');
                        if (!isCouponNotEmpty()) {
                            betCoupon.classList.remove('visible');
                        }
                    }
                });
            });

            document.querySelectorAll('.coupon-tab').forEach(tab => {
                tab.addEventListener('click', () => {
                    document.querySelectorAll('.coupon-tab').forEach(t => t.classList.remove('active'));
                    tab.classList.add('active');
                    const tabId = tab.getAttribute('data-tab');
                    document.getElementById('couponItems').style.display = tabId === 'bets' ? 'block' : 'none';
                    document.getElementById('couponHistory').style.display = tabId === 'history' ? 'block' : 'none';
                    document.querySelector('.coupon-total').style.display = tabId === 'bets' ? 'block' : 'none';
                    document.querySelector('.coupon-header').style.display = tabId === 'bets' ? 'flex' : 'none';
                    if (tabId === 'history') {
                        updateCouponHistory();
                    }
                });
            });

            clearCouponBtn.addEventListener('click', () => {
                clearCoupon();
                betCoupon.classList.remove('visible');
                document.querySelectorAll('.odds-item').forEach(item => {
                    item.classList.remove('selected');
                });
            });

            notificationsBtn.addEventListener('click', () => {
                showNotification('Уведомления пока не реализованы');
                notificationCount.textContent = '0';
            });

            function removeFromCouponByEvent(eventName, betType, matchId) {
                const index = couponItems.findIndex(item => 
                    item.event === eventName && item.bet === betType && item.matchId === matchId
                );
                if (index !== -1) {
                    removeFromCoupon(index);
                }
            }

            function isCouponNotEmpty() {
                return couponItems.length > 0;
            }

            function highlightSelectedOdds() {
                document.querySelectorAll('.odds-item').forEach(item => {
                    const event = item.getAttribute('data-event');
                    const bet = item.getAttribute('data-bet');
                    const matchId = item.getAttribute('data-match-id');
                    if (isInCoupon(event, bet, matchId)) {
                        item.classList.add('selected');
                        betCoupon.classList.add('visible');
                    }
                });
            }

            function showNotification(message) {
                let notification = document.getElementById('global-notification');
                if (!notification) {
                    notification = document.createElement('div');
                    notification.id = 'global-notification';
                    document.body.appendChild(notification);
                }
                notification.textContent = message;
                notification.style.opacity = '1';
                setTimeout(() => {
                    notification.style.opacity = '0';
                }, 3000);
            }

            function openProfile() {
                document.getElementById('profileModal').classList.add('show');
                updateBetHistory();
            }

            function closeProfile() {
                document.getElementById('profileModal').classList.remove('show');
            }

            function openHistory() {
                document.getElementById('profileModal').classList.remove('show');
                document.getElementById('historyModal').classList.add('show');
                updateBetHistory();
            }

            function closeHistory() {
                document.getElementById('historyModal').classList.remove('show');
            }

            highlightSelectedOdds();
            updateBalance();
            initCoupon();

            document.querySelectorAll('.date-arrow').forEach(arrow => {
                arrow.addEventListener('click', function() {
                    const currentTitle = document.querySelector('.date-title').textContent;
                    console.log('Переключение даты: ', currentTitle);
                });
            });
        });
    </script>
